### My Code

    class MinStack {

        Stack<Integer> stack = new Stack<Integer>();
        Stack<Integer> min = new Stack<Integer>();

        public void push(int x) {
            stack.push(x);
            if (min.isEmpty() || min.peek() >= x) {
                min.push(x);
            }
        }

        public void pop() {
            if (stack.isEmpty()) {
                return;
            }
            int topNum = stack.pop();
            if (topNum == min.peek()) {
                min.pop();
            }
        }

        public int top() {
            if (stack.isEmpty()) {
                return 0;
            }
            return stack.peek();
        }

        public int getMin() {
            if (min.isEmpty()) {
                return 0;
            }
            return min.peek();
        }
    }
