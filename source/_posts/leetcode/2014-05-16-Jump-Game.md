### My code


    public boolean canJump(int[] A) {
        int len = A.length, max = 0;
        if (len == 0) return true;
        for (int i = 0; i < len && i <= max; i ++) {
            max = Math.max(max, i + A[i]);
            if (max >= len - 1) return true;
        }
        return (max >= len - 1);
    }
